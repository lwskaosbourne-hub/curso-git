(Adiciona os arquivos ao commit)

git log --pretty=oneline

(Mostra o status do projeto)

git checkout -b "nome-da-branch"

(Cria uma nova branch fora da branch main)

git push --set-upstream origin nova-feature

(Faz um push para o repositório direto da feature atual)

git checkout main

(Volta para a branch main long-living)

git pull

(Atualiza as alterações do código/projeto)

git branch -d nova-feature

(Deleta a branch local nova-feature)

git stash

(Guarda alterações da branch em um "bolso" para poder dar um git pull)

git stash pop

(Recupera as alterações guardadas anteriormente pelo git stash)
